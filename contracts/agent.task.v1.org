** task-has-workflow-state
:PROPERTIES:
:ASSERT_ID: task.has-workflow-state
:SEVERITY: error
:END:

#+BEGIN_SRC org-contract
(assert count >= 1
  (and
    (headline :at $scope)
    (or
      (= (property :todo-type) "todo")
      (= (property :todo-type) "done"))))
#+END_SRC

#+BEGIN_SRC jinja2 :name message
Task `{{ scope.title }}` must be represented by an Org TODO workflow state.
#+END_SRC

** task-acceptance-has-checklist
:PROPERTIES:
:ASSERT_ID: task.acceptance-has-checklist
:SEVERITY: error
:END:

#+BEGIN_SRC org-contract
(let ((acceptance
       (headline :child-of $scope :summary (title "Acceptance"))))
  (assert count >= 1
    (and
      (item :descendant-of acceptance)
      (or
        (= (summary checkbox) "off")
        (= (summary checkbox) "on")
        (= (summary checkbox) "trans")))))
#+END_SRC

#+BEGIN_SRC jinja2 :name message
Task `{{ scope.title }}` Acceptance section must include checkbox criteria.
#+END_SRC
